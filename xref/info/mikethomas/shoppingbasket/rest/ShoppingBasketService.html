59  @Path("/shoppingbasket")
60  @Produces(value = {MediaType.APPLICATION_JSON, MediaType.APPLICATION_XML})
61  @Api(tags = "Shopping Basket")
62  public class ShoppingBasketService {
63  
64      protected static Map<Integer, Basket> baskets = new HashMap<>();
65      protected final Items list = getFromXml(Items.class, "Items.xml");
66  
67      @GET
68      @Path("/items")
69      @ApiResponses(value = {
70          @ApiResponse(code = 200, message = "List of items", response = Items.class)
71      })
72      public Response getItems() {
73          return Response.status(200).entity(list).build();
74      }
75  
76      @POST
77      @Path("/add/{basket_id}")
78      public Response addItem(
79              @PathParam("basket_id") int basketId,
80              @QueryParam("item") int itemId,
81              @QueryParam("quantity") int quantity) {
82  
83          baskets.putIfAbsent(basketId, new Basket());
84          Basket basket = baskets.get(basketId);
85          Item item = list.getItem().get(itemId);
86          basket.getBasket().putIfAbsent(item, quantity);
87          return Response.status(200).entity(basket).build();
88      }
89  
90      @PUT
91      @Path("/update/{basket_id}")
92      public Response replaceItem(
93              @PathParam("basket_id") int basketId,
94              @QueryParam("item") int itemId,
95              @QueryParam("quantity") int quantity) {
96  
97          baskets.putIfAbsent(basketId, new Basket());
98          Basket basket = baskets.get(basketId);
99          Item item = list.getItem().get(itemId);
100         basket.getBasket().replace(item, quantity);
101         return Response.status(200).entity(basket).build();
102     }
103 
104     @DELETE
105     @Path("/remove/{basket_id}")
106     public Response removeItem(
107             @PathParam("basket_id") int basketId,
108             @QueryParam("item") int itemId) {
109 
110         baskets.putIfAbsent(basketId, new Basket());
111         Basket basket = baskets.get(basketId);
112         Item item = list.getItem().get(itemId);
113         basket.getBasket().remove(item);
114         return Response.status(200).entity(basket).build();
115     }
116 
117     @GET
118     @Path("/contents/{basket_id}")
119     public Response getContents(
120             @PathParam("basket_id") int basketId) {
121 
122         baskets.putIfAbsent(basketId, new Basket());
123         Basket basket = baskets.get(basketId);
124         return Response.status(200).entity(basket).build();
125     }
126 
127     @GET
128     @Path("/total/{basket_id}")
129     public Response getTotal(
130             @PathParam("basket_id") int basketId) {
131 
132         baskets.putIfAbsent(basketId, new Basket());
133         Basket basket = baskets.get(basketId);
134         return Response.status(200).entity(basket.getTotal()).build();
135     }

137     /**
138      * <p>Helper method to construct the service classes from XML files.</p>
139      *
140      * @param type a {@link java.lang.Class} object.
141      * @param file a {@link java.lang.String} object.
142      * @param <T> a T object.
143      * @return a T object.
144      */
145     public static <T> T getFromXml(Class<T> type, String file) {
146         try {
147             JAXBContext context = JAXBContext.newInstance(type);
148             Unmarshaller unmarshaller = context.createUnmarshaller();
149             URL xml = type.getResource(file);
150             return type.cast(unmarshaller.unmarshal(xml));
151         } catch (JAXBException ex) {
152             System.err.println(ex);
153         }
154         return null;
155     }
